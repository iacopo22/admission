int counter = 0;

int main(void)
{

    int n = get_int("Integer: ");

    int number = collatz(n);
    printf("Number of times: %i\n", number);

int collatz(int n)
{
    //base case
    if (n == 1)
    {
        return 1;
    }
    else if (n % 2 == 0)
    {
        counter++;
        collatz(n / 2);
    }
    else if (n % 2 != 0)
    {
        counter++;
        collatz((3 * n) + 1);
    }
    return counter;
}
